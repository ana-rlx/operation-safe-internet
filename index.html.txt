  <script>
    // Quiz data
    const quizData = [
      { q: 'Which of these is private information you should NOT share?', options: ['Your favorite color','Your home address','The name of a cartoon you like'], a:1 },
      { q: 'What makes a strong password?', options: ['Short word','Four random words/passphrase','Your pet\'s name'], a:1 },
      { q: 'If you get an urgent message asking for your password, you should:', options: ['Reply with your password','Ask an adult and do not share your password','Click the link immediately'], a:1 },
      { q: 'What should you do about mean comments online?', options: ['Respond with mean words','Block/report and tell an adult','Share them with friends to laugh'], a:1 }
    ];

    function buildQuiz(){
      const area = document.getElementById('quizArea');
      area.innerHTML = '';
      quizData.forEach((item, idx)=>{
        const qDiv = document.createElement('div');
        qDiv.className = 'card';
        const h = document.createElement('h3'); h.textContent = (idx+1) + '. ' + item.q; qDiv.appendChild(h);
        item.options.forEach((opt,i)=>{
          const label = document.createElement('label');
          label.className = 'quiz-option';
          label.setAttribute('role','button');
          label.tabIndex = 0;
          label.innerHTML = `<input type="radio" name="q${idx}" value="${i}" style="margin-right:8px"> ${opt}`;
          qDiv.appendChild(label);
        });
        area.appendChild(qDiv);
      });
    }

    function gradeQuiz(){
      let score = 0;
      quizData.forEach((item, idx)=>{
        const selected = document.querySelector(`input[name=q${idx}]:checked`);
        const labels = document.getElementsByName(`q${idx}`);
        // mark answers visually
        const optionLabels = Array.from(document.querySelectorAll(`#quizArea .card`))[idx].querySelectorAll('.quiz-option');
        optionLabels.forEach((lab, i)=>{
          lab.classList.remove('correct','wrong');
          const input = lab.querySelector('input');
          if(i === item.a){ lab.classList.add('correct'); }
          if(input.checked && i !== item.a){ lab.classList.add('wrong'); }
        });
        if(selected && Number(selected.value) === item.a) score++;
      });
      document.getElementById('score').textContent = `Score: ${score} / ${quizData.length}`;
    }

    document.getElementById('submitQuiz').addEventListener('click', gradeQuiz);
    document.getElementById('printWorksheet').addEventListener('click', ()=>{
      const w = document.getElementById('worksheet').innerHTML;
      const newWin = window.open('', '_blank');
      newWin.document.write('<html><head><title>Worksheet</title></head><body>'+w+'</body></html>');
      newWin.document.close();
      newWin.print();
    });

    // allow clicking the whole label to select radio
    document.addEventListener('click', e=>{
      if(e.target.closest('.quiz-option')){
        const lab = e.target.closest('.quiz-option');
        const radio = lab.querySelector('input[type=radio]');
        if(radio) radio.checked = true;
      }
    });

    // build when loaded
    buildQuiz();

    // helper: replace the YouTube video by id (optional runtime change)
    function setYouTubeVideo(videoId){
      const iframe = document.getElementById('youtubeEmbed');
      if(!iframe) return;
      iframe.src = `https://www.youtube.com/embed/${videoId}`;
    }
